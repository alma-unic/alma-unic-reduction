211    def _cont_from_file(self) -> Dict[int, List[str]]:
212        """Read continuum ranges from `cont_file`."""
213        # Read file
214        with self.cont_file.open() as cfile:
215            lines = cfile.readlines()
216
217        # Find source and get continuum ranges
218        is_field = False
219        cont = {}
220        spw_orig = list(map(int, self.config['split']['spw'].split(',')))
221        for line in lines:
222            if line.startswith('Field') and self.name in line:
223                is_field = True
224                continue
225            elif line.startswith('Field') and is_field:
226                break
227            if is_field and 'SpectralWindow' in line:
228                number = int(line.split()[1])
229                spw = spw_orig.index(number)
230            elif '~' not in line:
231                continue
232            elif is_field and '~' in line:
233                if spw not in cont:
234                    cont[spw] = [line.split()[0]]
235                else:
236                    cont[spw].append(line.split()[0])
237
238        # Copy continuum ranges for multi EBs
239        final_cont = {}
240        for i, spws in enumerate(self.spws):
241            for spw in map(int, spws.split(',')):
242                final_cont[spw] = cont[i]
243        final_cont = dict(sorted(final_cont.items()))
244
245        return final_cont

247    def get_spectral_ranges(self,
248                            suffix: str,
249                            invert: bool = False,
250                            mask_borders: bool = False,
251                            resume: bool = False) -> str:
252        """Get spectral ranges from continuum file.
253        
254        The `cont.dat` contains the spectral ranges where the continuum is
255        calculated. The `invert` parameter allows to select the channels where
256        lines are.
257
258        Args:
259          suffix: File name suffix.
260          invert: Optional; Reverse channel selection?
261          mask_borders: Optional; Mask 10 channels at beginning and end?
262          reume: Optional; Resume where it was left last time?
263
264        Returns:
265          A CASA compatible `spw` selection.
266        """
267        flag_file = self.uvdata.with_suffix(f'{suffix}.json')
268        if flag_file.is_file() and resume:
269            flags = json.loads(flag_file.read_text())
270        else:
271            flags = self._cont_from_file()
272            self.log.debug('Flags (frequency): %s', flags)
273            flags = flags_from_cont_ranges(flags, self.uvdata, invert=invert,
274                                           mask_borders=mask_borders)
275            flag_file.write_text(json.dumps(flags, indent=4))
276        self.log.debug('Flags (channels): %s', flags)
277
278        return flags

280    def get_image_scales(self,
281                         uvtype: str = '',
282                         spw: Optional[int] = None,
283                         beam_oversample: int = 5,
284                         sigfig: int = 2,
285                         pbcoverage: float = 2.) -> Tuple[str, int]:
286        """Get cell and image sizes.
287        
288        Args:
289          uvtype: Optional; Type of uvdata to image.
290          spw: Optional; Select spectral window.
291          beam_oversample: Optional; Number of pixels per beam.
292          sigfig: Optional; Significant figures of the pixel size.
293          pbcoverage: Optional; Number of PB sizes for the image size.
294
295        Returns:
296          CASA compatible cell size and image size.
297        """
298        # Get parameters from MS
299        uvdata = self.get_uvname(uvtype)
300        low_freq, high_freq, longest_baseline = extrema_ms(uvdata, spw=spw)
301        self.log.info('Frequency range: %s - %s', low_freq, high_freq)
302        self.log.info('Longest baseline: %s', longest_baseline)
303
304        # Size
305        beam = gaussian_beam(high_freq, longest_baseline)
306        pbsize = gaussian_primary_beam(low_freq, u.Quantity(self.array))
307        self.log.info('Estimated smallest beam size: %s', beam)
308        self.log.info('Estimated largest pb size: %s', pbsize)
309
310        # Estimate cell size
311        su = synthesisutils()
312        cell = round_sigfig(beam / beam_oversample, sigfig=sigfig)
313        self.log.info('Estimated cell size: %s', cell)
314        imsize = int(pbsize * pbcoverage / cell)
315        self.log.info('Estimated image size: %s', imsize)
316        imsize = su.getOptimumSize(imsize)
317        self.log.info('Optimal image size: %s', imsize)
318
319        return f'{cell.value}{cell.unit}', imsize

357                      section: str,
358                      outdir: Path = Path('./'),
359                      uvtype: str = '',
360                      nproc: int = 5,
361                      resume: bool = False,
362                      **tclean_args):
363        """Run `tclean` on all spws.
364        
365        Arguments given under `tclean_args` replace the values from `config`.
366
367        Args:
368          section: Configuration section.
369          outdir: Output directory path.
370          uvtype: Optional; Type of uvdata to clean.
371          nproc: Optional; Number of processors for parallel clean.
372          resume: Optional; Resume where it was left?
373          tclean_args: Optional; Additional arguments for tclean.
374        """
375        # Iterate over spws
376        for i, spw in enumerate(self.spws):
377            # Clean args
378            suffix = self.get_uvsuffix(uvtype)
379            imagename = outdir / (f'{self.name}_{self.array}_spw{i}'
380                                  f'{suffix}.image')
381
382            # Check for files
383            if imagename.exists() and resume:
384                self.log.info('Skipping cube for spw %i', i)
385                continue
386            elif imagename.exists() and not resume:
387                self.log.warning('Deleting cubes for spw %i', i)
388                os.system(f"rm -rf {imagename.with_suffix('.*')}")
389
390            # Parameters
391            cell = self.config.get(section, 'cell', vars=tclean_args,
392                                   fallback=None)
393            imsize = self.config.get(section, 'imsize', vars=tclean_args,
394                                     fallback=None)
395            impars = {}
396            if cell is None or imsize is None:
397                new_cell, new_imsize = self.get_image_scales(spw=i)
398                if cell is None:
399                    impars['cell'] = new_cell
400                if imsize is None:
401                    impars['imsize'] = f'{new_imsize}'
402
403            # Run clean
404            self.clean_data(section, imagename, uvtype=uvtype, nproc=nproc,
405                            resume=resume, spw=spw, **tclean_args, **impars)
406            print('-' * 80)

408    def contsub(self,
409                section: str = 'contsub',
410                resume: bool = False) -> None:
411        """Run `uvcontsub` to obtain continuum subtracted visibilities.
412        
413        Args:

416        """
417        val_step = validate_step(resume, self.uvcontsub, log=self.log.warning)
418        if not val_step:
419            return

421        # Prepare input
422        suffix = self.get_uvsuffix('uvcontsub')
423        mask_borders = self.config.getboolean(section, 'mask_borders')
424        fitspec = self.get_spectral_ranges(suffix, invert=True,

429        # Run uvcontsub
430        uvcontsub(vis=f'{self.uvdata}', outputvis=f'{self.uvcontsub}',
431                  fitspec=clumps_to_casa(fitspec), fitorder=fitorder)

433    def continuum_vis(self,
434                      section: str = 'continuum',
435                      resume: bool = False) -> Tuple[Path]:
436        """Generate continuum visibilities.
437
438        It generates 2 visibility sets:
439
440        - Average of all the channels using the input width.
441        - Average of the line free channels using the `cont_file` and width.
442
443        Args:
444          section: Optional; Configuration section.
445          resume: Optional; Continue from where it was left?
446
447        Returns:
448          The line-free and averaged MS file names.
449        """
450        # Check cont_file
451        if self.cont_file is None:
452            self.log.warning('Continuum file not assigned, nothing to do')
453            return
454
455        # Check files
456        allchans = self.uvdata.with_suffix('.cont_all.ms')
457        linefree = self.uvcontinuum
458        val_all = validate_step(resume, allchans, log=self.log.warning)
459        val_lf = validate_step(resume, linefree, log=self.log.warning)
460
461        # Values from config
462        width = self.config.get(section, 'width', fallback=None)
463        if width is not None:
464            width = list(map(int, width.split(',')))
465        else:
466            width = continuum_bins(self.uvdata, u.Quantity(self.array))
467            self.config[section]['width'] = ','.join(map(str, width))
468            self.write_config()
469        self.log.info('Continuum bins: %s', width)
470        datacolumn = self.config[section]['datacolumn']
471
472        if val_all:
473            # Average channels without flagging
474            split_ms(vis=f'{self.uvdata}', outputvis=f'{allchans}', width=width,
475                     datacolumn=datacolumn)
476        if val_lf:
477            # Get line free channels
478            suffix = self.get_uvsuffix('continuum')
479            flag_chans = self.get_spectral_ranges(suffix,
480                                                  resume=resume)
481            casa_flag_chans = clumps_to_casa(flag_chans)
482
483            # Split the uvdata
484            # WARNING: this needs to be tested
485            # otherwise, use flagmanager and then split the visibilities
486            flagmanager(vis=f'{self.uvdata}', mode='save',
487                        versionname='before_cont_flags')
488            flagdata(vis=f'{self.uvdata}', mode='manual', spw=casa_flag_chans,
489                     flagbackup=False)
490            split_ms(vis=f'{self.uvdata}', outputvis=f'{linefree}',
491                     width=width, datacolumn=datacolumn)
492            flagmanager(vis=f'{self.uvdata}', mode='restore',
493                        versionname='before_cont_flags')
494
495        return linefree, allchans

554    def append_data(self,
555                    original_uvdata: Path,
556                    field: str,
557                    default_config: Path,
558                    configfile: Path,
559                    cont_file: Path,
560                    datadir: Path):
561        """Append data from original calibrated MS.
562        
563        Args:
564          original_uvdata: Original uv data.

566          default_config:
567        """
568        # Store original data for each array and calculate the number of EBs
569        array = get_array(original_uvdata)
570        self.log.info('Data array: %s', array)
571        self.original[array] += (original_uvdata,)
572        self.log.debug('Original data: %s', self.original)
573        neb = len(self.original[array])

575        # Update or create new FieldHandler
576        if (handler:= self.data_handler.get(array)) is None:
577            self.log.debug('Generating new data handler')
578            self._init_data_handler(field,
579                                    array,
580                                    default_config=default_config,
581                                    configfile=configfile,
582                                    cont_file=cont_file,
583                                    datadir=datadir,
584                                    neb=neb)
585        else:
586            self.log.debug('Updating data handler')
587            handler.neb = neb
588            handler.update_config(DEFAULT={'neb': f'{neb}'})
589
590        # Split the data from original MS?
591        split = self.datadir(array) / f'{self.name}_{array}_eb{neb}.ms'
592        if split.exists() and not self.resume:
593            self.log.warning('Deleting split ms: %s', split)
594            os.system(f'rm -rf {split}')
595        if not split.exists():
596            self.log.info('Splitting input ms to: %s', split)
597            self.split_field(array, original_uvdata, split)
598        else:
599            self.log.info('Found split ms: %s', split)
600        self.split[array] += (split,)
601        self.log.debug('Split data: %s', self.split)

212    def _cont_from_file(self) -> Dict[int, List[str]]:
213        """Read continuum ranges from `cont_file`."""
214        # Read file

240        final_cont = {}
241        for i, spws in enumerate(self.spws):
242            for spw in map(int, spws.split(',')):
243                final_cont[spw] = cont[i]
