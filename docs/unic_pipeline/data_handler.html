 123        elif self.array != self.config['DEFAULT']['array']:

 139                pass

 230    def _cont_from_file(self) -> Dict[int, List[str]]:
 231        """Read continuum ranges from `cont_file`."""
 232        # Read file
 233        with self.cont_file.open() as cfile:
 234            lines = cfile.readlines()
 235
 236        # Find source and get continuum ranges
 237        is_field = False
 238        cont = {}
 239        spw_orig = list(map(int, self.config['split']['spw'].split(',')))
 240        for line in lines:
 241            if line.startswith('Field') and self.name in line:
 242                is_field = True
 243                continue
 244            elif line.startswith('Field') and is_field:
 245                break
 246            if is_field and 'SpectralWindow' in line:
 247                number = int(line.split()[1])
 248                spw = spw_orig.index(number)
 249            elif '~' not in line:
 250                continue
 251            elif is_field and '~' in line:
 252                if spw not in cont:
 253                    cont[spw] = [line.split()[0]]
 254                else:
 255                    cont[spw].append(line.split()[0])
 256
 257        # Copy continuum ranges for multi EBs
 258        final_cont = {}
 259        for i, spws in enumerate(self.spws):
 260            for spw in map(int, spws.split(',')):
 261                final_cont[spw] = cont[i]
 262        final_cont = dict(sorted(final_cont.items()))
 263
 264        return final_cont

 266    def get_spectral_ranges(self,
 267                            suffix: str,
 268                            invert: bool = False,
 269                            mask_borders: bool = False,
 270                            resume: bool = False) -> Dict[int,
 271                                                          List[Tuple[int]]]:
 272        """Get spectral ranges from continuum file.
 273        
 274        The `cont.dat` contains the spectral ranges where the continuum is
 275        calculated. The `invert` parameter allows to select the channels where
 276        lines are.
 277
 278        Args:
 279          suffix: File name suffix.
 280          invert: Optional; Reverse channel selection?
 281          mask_borders: Optional; Mask 10 channels at beginning and end?
 282          reume: Optional; Resume where it was left last time?
 283
 284        Returns:
 285          A dictionary mapping SPW numbers with channel ranges.
 286        """
 287        flag_file = self.uvdata.with_suffix(f'{suffix}.json')
 288        if flag_file.is_file() and resume:
 289            flags = json.loads(flag_file.read_text())
 290        else:
 291            flags = self._cont_from_file()
 292            self.log.debug('Flags (frequency): %s', flags)
 293            flags = flags_from_cont_ranges(flags, self.uvdata, invert=invert,
 294                                           mask_borders=mask_borders)
 295            flag_file.write_text(json.dumps(flags, indent=4))
 296        self.log.debug('Flags (channels): %s', flags)
 297
 298        return flags

 300    def get_image_scales(self,
 301                         uvtype: str = '',
 302                         spw: Optional[int] = None,
 303                         beam_oversample: int = 5,
 304                         sigfig: int = 2,
 305                         pbcoverage: float = 2.,
 306                         section: str = 'imaging') -> Tuple[str, int]:
 307        """Get cell and image sizes.
 308        
 309        Args:
 310          uvtype: Optional; Type of uvdata to image.
 311          spw: Optional; Select spectral window.
 312          beam_oversample: Optional; Number of pixels per beam.
 313          sigfig: Optional; Significant figures of the pixel size.
 314          pbcoverage: Optional; Number of PB sizes for the image size.
 315
 316        Returns:
 317          CASA compatible cell size and image size.
 318        """
 319        # Search in config
 320        cell = self.config.get(section, 'cell', fallback=None)
 321        imsize = self.config.getint(section, 'imsize', fallback=None)
 322        if cell is not None and imsize is not None:
 323            return cell, imsize
 324
 325        # Get parameters from MS
 326        uvdata = self.get_uvname(uvtype)
 327        low_freq, high_freq, longest_baseline = extrema_ms(uvdata, spw=spw)
 328        self.log.info('Frequency range: %s - %s', low_freq, high_freq)
 329        self.log.info('Longest baseline: %s', longest_baseline)
 330
 331        # Size
 332        antenna = min(self.get_antenna_diams(self.array))
 333        beam = gaussian_beam(high_freq, longest_baseline)
 334        pbsize = gaussian_primary_beam(low_freq, antenna)
 335        self.log.info('Estimated smallest beam size: %s', beam)
 336        self.log.info('Estimated largest pb size: %s', pbsize)
 337
 338        # Estimate cell size
 339        su = synthesisutils()
 340        if cell is None:
 341            cell = round_sigfig(beam / beam_oversample, sigfig=sigfig)
 342            self.log.info('Estimated cell size: %s', cell)
 343            cell = f'{cell.value}{cell.unit}'
 344            self.config[section]['cell'] = cell
 345            self.write_config()
 346        if imsize is None:
 347            imsize = int(pbsize * pbcoverage / cell)
 348            self.log.info('Estimated image size: %s', imsize)
 349            imsize = su.getOptimumSize(imsize)
 350            self.log.info('Optimal image size: %s', imsize)
 351            self.config[section]['imsize'] = imsize
 352            self.write_config()
 353
 354        return cell, imsize

 357                   section: str,
 358                   imagename: Path,
 359                   uvtype: str,
 360                   nproc: int = 5,
 361                   resume: bool = False,
 362                   **tclean_args):
 363        """Run tclean on input or stored uvdata.
 364
 365        Arguments given under `tclean_args` replace the values from `config`.
 366
 367        Args:
 368          section: Configuration section.
 369          imagename: Image directory path.
 370          uvtype: Type of uvdata to clean.
 371          nproc: Optional; Number of processors for parallel clean.
 372          resume: Optional; Resume where it was left?
 373          tclean_args: Optional; Additional arguments for tclean.
 374        """
 375        # tclean args
 376        config = self.config[section]
 377        kwargs = get_tclean_params(config, cfgvars=tclean_args)
 378        if uvtype == 'continuum':
 379            kwargs.setdefault('specmode', 'mfs')
 380        elif uvtype == 'uvcontsub':
 381            kwargs.setdefault('specmode', 'cube')
 382            kwargs.setdefault('outframe', 'LSRK')
 383        kwargs.setdefault('deconvolver', 'hogbom')
 384        kwargs.setdefault('weighting', 'briggs')
 385        kwargs.setdefault('robust', 0.5)
 386        kwargs.setdefault('gridder', 'standard')
 387        kwargs.setdefault('niter', 100000)
 388
 389        # Masking
 390        if kwargs.get('usemask', '') == 'auto-multithresh':
 391            kwargs = recommended_auto_masking(self.array) | kwargs
 392
 393        # Check for files
 394        if imagename.exists() and resume:
 395            return
 396        elif imagename.exists() and not resume:
 397            self.log.warning('Deleting image: %s', imagename)
 398            os.system(f"rm -rf {imagename.with_suffix('.*')}")
 399
 400        # Run tclean
 401        uvdata = self.get_uvname(uvtype)
 402        tclean_parallel(uvdata, imagename.with_name(imagename.stem),
 403                        nproc, kwargs, log=self.log)

 514    def continuum_vis(self,
 515                      section: str = 'continuum',
 516                      resume: bool = False,
 517                      plot_selection: bool = False) -> Tuple[Path]:
 518        """Generate continuum visibilities.
 519
 520        It generates 2 visibility sets:
 521
 522        - Average of all the channels using the input width.
 523        - Average of the line free channels using the `cont_file` and width.
 524
 525        Args:
 526          section: Optional; Configuration section.
 527          resume: Optional; Continue from where it was left?
 528          plot_selection: Optional; Plot selection for flagged channels?
 529
 530        Returns:
 531          The line-free and averaged MS file names.
 532        """
 533        # Check cont_file
 534        if self.cont_file is None:
 535            self.log.warning('Continuum file not assigned, nothing to do')
 536            return
 537
 538        # Check files
 539        allchans = self.uvdata.with_suffix('.cont_all.ms')
 540        linefree = self.uvcontinuum
 541        val_all = validate_step(resume, allchans, log=self.log.warning)
 542        val_lf = validate_step(resume, linefree, log=self.log.warning)
 543
 544        # Values from config
 545        width = self.config.get(section, 'width', fallback=None)
 546        if width is not None:
 547            width = list(map(int, width.split(',')))
 548        else:
 549            width = continuum_bins(self.uvdata, u.Quantity(self.array))
 550            self.config[section]['width'] = ','.join(map(str, width))
 551            self.write_config()
 552        self.log.info('Continuum bins: %s', width)
 553        datacolumn = self.config.get(section, 'datacolumn', fallback='data')
 554
 555        if val_all:
 556            # Average channels without flagging
 557            split_ms(vis=f'{self.uvdata}', outputvis=f'{allchans}', width=width,
 558                     datacolumn=datacolumn)
 559        if val_lf:
 560            # Get line free channels
 561            suffix = self.get_uvsuffix('continuum')
 562            mask_borders = self.config.getboolean(section, 'mask_borders')
 563            flag_chans = self.get_spectral_ranges(suffix,
 564                                                  mask_borders=mask_borders,
 565                                                  resume=resume)
 566            casa_flag_chans = clumps_to_casa(flag_chans)
 567
 568            # Plot?
 569            if plot_selection and len(self.spectra) != 0:
 570                self.log.info('Plotting spectra')
 571                outdir = linefree.parent / 'plots'
 572                outdir.mkdir(exist_ok=True)
 573                plot_spectral_selection(self.spectra, flag_chans, self.uvdata,
 574                                        self.spws, outdir, suffix=suffix)
 575            elif plot_selection:
 576                self.log.info('No spectra to plot')
 577
 578            # Split the uvdata
 579            # WARNING: this needs to be tested
 580            # otherwise, use flagmanager and then split the visibilities
 581            flagmanager(vis=f'{self.uvdata}', mode='save',
 582                        versionname='before_cont_flags')

232        """Read continuum ranges from `cont_file`."""
233        # Read file
234        with self.cont_file.open() as cfile:
235            lines = cfile.readlines()
236
237        # Find source and get continuum ranges
238        is_field = False
239        cont = {}
240        spw_orig = list(map(int, self.config['split']['spw'].split(',')))
241        for line in lines:
242            if line.startswith('Field') and self.name in line:
243                is_field = True
244                continue
245            elif line.startswith('Field') and is_field:
246                break
247            if is_field and 'SpectralWindow' in line:
248                number = int(line.split()[1])
249                spw = spw_orig.index(number)
250            elif '~' not in line:
251                continue
252            elif is_field and '~' in line:
253                if spw not in cont:
254                    cont[spw] = [line.split()[0]]
255                else:
256                    cont[spw].append(line.split()[0])
257
258        # Copy continuum ranges for multi EBs
259        final_cont = {}
260        for i, spws in enumerate(self.spws):
261            for spw in map(int, spws.split(',')):
262                final_cont[spw] = cont[i]
263        final_cont = dict(sorted(final_cont.items()))
264
265        return final_cont
